🔗 Logical Operators - AND, OR, NOT

Logical operators allow you to combine multiple conditions and create sophisticated decision-making logic in your programs.

🔗 AND Operator

# Both conditions must be True
age = 20
has_license = True

if age >= 18 and has_license:
    print("Can drive! 🚗")
else:
    print("Cannot drive yet")

# Multiple AND conditions
score = 85
attendance = 95
if score >= 80 and attendance >= 90:
    print("Honor roll! 🏆")

🔀 OR Operator

# At least one condition must be True
weekend = True
holiday = False

if weekend or holiday:
    print("No work today! 🎉")
else:
    print("Work day 💼")

# Emergency conditions
fire = False
earthquake = False
if fire or earthquake:
    print("EVACUATE! 🚨")

❌ NOT Operator

# Reverses the boolean value
is_raining = False

if not is_raining:
    print("Perfect day for a picnic! ☀️")
else:
    print("Stay inside 🌧️")

# Checking for invalid input
password = "abc123"
if not (len(password) >= 8):
    print("Password too short!")

🔑 Truth Table Reference

• **True and True** = True
• **True and False** = False
• **True or False** = True
• **False or False** = False
• **not True** = False
• **not False** = True

🎯 Grade Calculator

def calculate_grade(score, attendance, participation):
    # Complex grading logic
    if score >= 90 and attendance >= 95:
        return "A+"
    elif score >= 85 and (attendance >= 90 or participation >= 80):
        return "A"
    elif score >= 80 and attendance >= 85:
        return "B+"
    elif score >= 75 or (attendance >= 95 and participation >= 85):
        return "B"
    elif score >= 70 and not (attendance < 80 and participation < 60):
        return "C"
    else:
        return "F"

# Test the function
print(calculate_grade(88, 92, 85)) # A

🌤️ Weather Advisory

def weather_advisory(temp, humidity, wind_speed, precipitation):
    advisories = []

    # Temperature warnings
    if temp > 100 or (temp > 95 and humidity > 60):
        advisories.append("🔥 Heat Warning")
    elif temp < 32 and wind_speed > 15:
        advisories.append("🧊 Wind Chill Alert")

    # Storm conditions
    if wind_speed > 40 and precipitation > 0.5:
        advisories.append("⛈️ Severe Storm")
    elif wind_speed > 25 or precipitation > 1.0:
        advisories.append("🌧️ Weather Advisory")

    # Perfect conditions
    if (70 <= temp <= 80 and humidity < 50 and
        wind_speed < 10 and precipitation == 0):
        advisories.append("☀️ Perfect Weather")

    return advisories if advisories else ["🌤️ Normal Conditions"]

🎯 Game State Management

class GameState:
    def __init__(self):
        self.player_health = 100
        self.player_score = 0
        self.level = 1
        self.inventory = []

    def take_damage(self, damage):
        self.player_health -= damage
        if self.player_health <= 0:
            return "Game Over"
        return "Continue"

    def add_score(self, points):
        self.player_score += points
        if self.player_score >= self.level * 1000:
            self.level += 1
            return "Level Up!"

🎮 Game Design Principles

• **Randomness** creates replayability and excitement
• **State management** tracks player progress
• **Feedback loops** reward player actions
• **Progressive difficulty** keeps players engaged

🎮 Input Handling

def get_player_choice(options):
    """Get valid player input"""
    while True:
        print("\\nChoose an action:")
        for i, option in enumerate(options, 1):
            print(f"{i}. {option}")

        try:
            choice = int(input("Enter choice: "))
            if 1 <= choice <= len(options):
                return choice - 1
            else:
                print("Invalid choice!")
        except ValueError:
            print("Please enter a number!")

🏆 Scoring Systems

class ScoreManager:
    def __init__(self):
        self.score = 0
        self.multiplier = 1
        self.combo = 0

    def add_points(self, base_points, bonus=False):
        if bonus:
            self.combo += 1
            self.multiplier = min(5, 1 + self.combo * 0.5)
        else:
            self.combo = 0
            self.multiplier = 1

        points = int(base_points * self.multiplier)
        self.score += points
        return points
